def predict_new_image(img):
    # Redimensionar la imagen a 64x64 (tamao usado para entrenamiento)
    img = cv2.resize(img, (64, 64))

    # Normalizar los valores de la imagen
    img = img.astype('float32') / 255.0

    # Expandir dimensiones para que coincidan con las del batch de entrenamiento (1, 64, 64, 3)
    img = np.expand_dims(img, axis=0)

    # Realizar la prediccin usando el modelo entrenado (cnn_v1)
    prediccion = cnn_v1.predict(img)

    # Obtener el ndice de la clase con mayor probabilidad
    pred_out = np.argmax(prediccion, axis=1)[0]  # np.argmax devuelve el ndice de la mayor probabilidad

    # Imprimir la clase correspondiente
    print(f"Prediccin: {arr_vals[pred_out]}")

    return arr_vals[pred_out]

import matplotlib.pyplot as plt

def compare_histories(histories, names):
    # Graficar la precisin
    plt.figure(figsize=(14, 5))

    # Subplot para la precisin
    plt.subplot(1, 2, 1)
    for history, name in zip(histories, names):
        plt.plot(history.history['accuracy'], label=f'{name} Entrenamiento')
        plt.plot(history.history['val_accuracy'], label=f'{name} Validacin')
    plt.title('Precisin del modelo')
    plt.xlabel('pocas')
    plt.ylabel('Precisin')
    plt.legend()

    # Subplot para la prdida
    plt.subplot(1, 2, 2)
    for history, name in zip(histories, names):
        plt.plot(history.history['loss'], label=f'{name} Entrenamiento')
        plt.plot(history.history['val_loss'], label=f'{name} Validacin')
    plt.title('Prdida del modelo')
    plt.xlabel('pocas')
    plt.ylabel('Prdida')
    plt.legend()

    plt.tight_layout()
    plt.show()

# Lista de historias y nombres de los modelos
